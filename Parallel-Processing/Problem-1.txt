%%writefile phonebook.cu
#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <cuda_runtime.h>

using namespace std;

// CUDA kernel: each thread checks one line
__global__ void searchKernel(char* data, int* lineOffsets, int numLines, char* searchKey, int keyLen, int* results) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < numLines) {
        char* line = data + lineOffsets[idx];
        // Simple substring search
        for (int i = 0; line[i] != '\0'; i++) {
            bool match = true;
            for (int j = 0; j < keyLen; j++) {
                if (line[i+j] == '\0' || line[i+j] != searchKey[j]) {
                    match = false;
                    break;
                }
            }
            if (match) {
                results[idx] = 1;
                return;
            }
        }
    }
}

int main(int argc, char* argv[]) {
    if (argc < 4) {
        cerr << "Usage: ./output <search_key> <num_threads> <file_path>" << endl;
        return 1;
    }

    string searchKey = argv[1];
    int numThreads   = stoi(argv[2]);   // GPU threads per block
    string filePath  = argv[3];

    // Load file into memory
    ifstream file(filePath);
    if (!file.is_open()) {
        cerr << "Cannot open file" << endl;
        return 1;
    }

    vector<string> lines;
    string line;
    while (getline(file, line)) {
        lines.push_back(line);
    }
    file.close();

    int numLines = lines.size();

    // Flatten lines into one big char array
    string allData;
    vector<int> lineOffsets;
    for (auto& l : lines) {
        lineOffsets.push_back(allData.size());
        allData += l + '\0';
    }

    // Allocate GPU memory
    char* d_data; int* d_offsets; char* d_key; int* d_results;
    cudaMalloc(&d_data, allData.size());
    cudaMalloc(&d_offsets, numLines * sizeof(int));
    cudaMalloc(&d_key, searchKey.size());
    cudaMalloc(&d_results, numLines * sizeof(int));

    // Copy data to GPU
    cudaMemcpy(d_data, allData.c_str(), allData.size(), cudaMemcpyHostToDevice);
    cudaMemcpy(d_offsets, lineOffsets.data(), numLines * sizeof(int), cudaMemcpyHostToDevice);
    cudaMemcpy(d_key, searchKey.c_str(), searchKey.size(), cudaMemcpyHostToDevice);
    cudaMemset(d_results, 0, numLines * sizeof(int));

    // Timing setup
    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);

    // Launch kernel
    int blocks = (numLines + numThreads - 1) / numThreads;
    cudaEventRecord(start);
    searchKernel<<<blocks, numThreads>>>(d_data, d_offsets, numLines, d_key, searchKey.size(), d_results);
    cudaEventRecord(stop);
    cudaEventSynchronize(stop);

    // Measure time
    float milliseconds = 0;
    cudaEventElapsedTime(&milliseconds, start, stop);

    // Copy results back
    vector<int> results(numLines);
    cudaMemcpy(results.data(), d_results, numLines * sizeof(int), cudaMemcpyDeviceToHost);

    // Print matches
    int matchCount = 0;
    for (int i = 0; i < numLines; i++) {
        if (results[i]) {
            cout << "Match: " << lines[i] << endl;
            matchCount++;
        }
    }
    if (matchCount == 0) cout << "No data loaded" << endl;

    // Log summary
    cout << "\n--- Execution Log ---" << endl;
    cout << "Total contents (lines): " << numLines << endl;
    cout << "GPU cores (threads per block): " << numThreads << endl;
    cout << "Kernel execution time: " << milliseconds << " ms" << endl;
    cout << "Entries found: " << matchCount << endl;

    // Free GPU memory
    cudaFree(d_data);
    cudaFree(d_offsets);
    cudaFree(d_key);
    cudaFree(d_results);

    return 0;
}
