%%writefile phonebook_mpi.cpp
#include <mpi.h>
#include <iostream>
#include <fstream>
#include <vector>
#include <string>

using namespace std;

int main(int argc, char* argv[])
{
    MPI_Init(&argc, &argv);

    int rank, size;

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    if(argc < 4)
    {
        if(rank == 0)
        {
            cout << "Usage: mpirun -np <processes> ./phonebook_mpi <search_string> <output_file> <input_files...>" << endl;
        }

        MPI_Finalize();
        return 0;
    }

    string searchStr = argv[1];
    string outputFile = argv[2];

    double startTime = MPI_Wtime();

    vector<string> localMatches;

    // Distribute files among processes
    for(int i = 3; i < argc; i++)
    {
        int fileIndex = i - 3;

        if(fileIndex % size != rank)
            continue;

        ifstream fin(argv[i]);

        if(!fin.is_open())
        {
            cerr << "Process " << rank
                 << " cannot open file: "
                 << argv[i] << endl;
            continue;
        }

        string line;

        while(getline(fin, line))
        {
            if(line.find(searchStr) != string::npos)
            {
                localMatches.push_back(line);
            }
        }

        fin.close();
    }

    string localResult;

    for(size_t i = 0; i < localMatches.size(); i++)
    {
        localResult += localMatches[i] + "\n";
    }

    int localLength = localResult.length();

    vector<int> recvCounts(size);

    MPI_Gather(
        &localLength,
        1,
        MPI_INT,
        recvCounts.data(),
        1,
        MPI_INT,
        0,
        MPI_COMM_WORLD);

    vector<int> displs(size);
    int totalLength = 0;

    if(rank == 0)
    {
        displs[0] = 0;

        for(int i = 1; i < size; i++)
        {
            displs[i] = displs[i - 1] + recvCounts[i - 1];
        }

        for(int i = 0; i < size; i++)
        {
            totalLength += recvCounts[i];
        }
    }

    vector<char> recvBuffer;

    if(rank == 0)
    {
        recvBuffer.resize(totalLength);
    }

    MPI_Gatherv(
        (void*)localResult.c_str(),
        localLength,
        MPI_CHAR,
        rank == 0 ? recvBuffer.data() : nullptr,
        recvCounts.data(),
        displs.data(),
        MPI_CHAR,
        0,
        MPI_COMM_WORLD);

    double endTime = MPI_Wtime();

    if(rank == 0)
    {
        ofstream fout(outputFile);

        for(int i = 0; i < totalLength; i++)
        {
            fout << recvBuffer[i];
        }

        fout.close();

        cout << "==================================" << endl;
        cout << "MPI Phonebook Search Completed" << endl;
        cout << "Processes Used : " << size << endl;
        cout << "Search String  : " << searchStr << endl;
        cout << "Execution Time : "
             << (endTime - startTime)
             << " seconds" << endl;
        cout << "Output File    : "
             << outputFile << endl;
        cout << "==================================" << endl;
    }

    MPI_Finalize();

    return 0;
}
